def connect_to_gnu(user, password, dbname, hostname, port):
    """
    Establece una conexión a una base de datos GNU Health.

    Args:
        user (str): Nombre de usuario para la conexión.
        password (str): Contraseña para la conexión.
        dbname (str): Nombre de la base de datos.
        hostname (str): Nombre del host o dirección IP del servidor.
        port (str): Puerto para la conexión.

    Returns:
        config: Objeto de configuración para la conexión establecida.
    """
    health_server = 'http://' + user + ':' + password + '@' + hostname + ':' + port + '/' + dbname + '/'
    print(f"Intentando conectar a {health_server}")

    try:
        print(health_server)
        conf = config.set_xmlrpc(health_server)

        print(conf)

        print(f"Conectado a la configuración")
        return conf
    except Exception as e:
        print(f"Error de conexión: {str(e)}")
